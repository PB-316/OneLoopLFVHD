Width decay of \(h \to \nu_i N_j\)¶

\begin{equation} \begin{aligned} \Gamma\left(h \rightarrow v N \right) &=\frac{Y_N^2}{8 \pi m_h^3} (m_h^2 - m_N^2)^2 \end{aligned} \end{equation}

from mpmath import *

mp.dps = 80; mp.pretty = True

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline

def Wh_nuN(mh,Mj,Cij):
    g = (2*80.379)/(246)
    mW = 80.379
    if Mj<mh:
        out = (Cij**2/(8*mp.pi*mh**3))*(mh**2 - Mj**2)**2
    else:
        out = 0.0
    return out

def WhnuN_sum(m6):
    out = 0
    for i in range(3):
        for j in range(3):
            out += Wh_nuN(125.1,mn(m6)[j],Ynuij(m6,i,j))
    return out
